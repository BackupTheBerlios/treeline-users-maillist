Hello:

I am attempting to make a plugin that pops up a dialog box I can interact 
with while still being able to interact with the main Treeline window.
I read in the PyQt documentation this is referred to as a 'modeless' dialog.

The best I've been able to accomplish is to get the dialog to appear for a 
split second before it abruptly disappear.
It looks like my dialog is appearing fine (I can see its outline and just 
make out the title) and is then destroyed.

Is a modeless plugin possible?  I'm new to python & Qt so perhaps I'm 
making a very obvious mistake?
I've tried numerous variations on the below code, all based upon using 
show(), but to no avail:

#!/usr/bin/env python
# Modeless Dialog Test
from PyQt4 import QtCore, QtGui

class modeless:

     # Add link to the menu
     def __init__(self, interface):
         self.interface = interface
         text = "Launch modeless dialog ..."
         menu = self.interface.getPulldownMenu(0)
         actionBefore = menu.actions()[2]
         action = QtGui.QAction(text, self.interface.mainWin)
         action.connect(action, QtCore.SIGNAL('triggered(bool)'), 
self.showWindow)
         menu.insertAction(actionBefore, action)

    # Display the modeless dialog
     def showWindow(self):
         window = MyForm()
         window.show()

class MyForm(QtGui.QWidget):

     # Setup the dialog
     def __init__(self, parent=None):
         QtGui.QWidget.__init__(self, parent)
         self.ui = Ui_Dialog()
         self.ui.setupUi(self)

class Ui_Dialog(object):

     # Customize the dialog
     def setupUi(self, Dialog):
         Dialog.setObjectName("Dialog")
         Dialog.resize(200, 200)

def main(interface):
     return modeless(interface)
